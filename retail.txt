<script>
    (function(window, $, components) {
        'use strict';

        var $timezonePopup = $("#economicCurrentTimePop");
        var $scrollableList = $timezonePopup.find('.js-scrollable-block');
        var $timezone = $("#timeZone");
        var $clock = $('#currentTime');
        var $timezoneOffset = $('#timeZoneGmtOffsetFormatted');
        var $options = $scrollableList.find('li');
        var $currentOption;

        var timezonePopup;
        var autoUpdateMode = true;
        var clockDateTime;
        var lastLocalClock;
        var timerId;
        var currentValue = $timezone.val();
        var optionsIdsPrefix = 'liTz';
        var callbacks = [];

        $(document).ready(function() {
            timezonePopup = new components.TooltipPopup($timezonePopup);
        });

        function makeDoubleCharsNumber(num) {
            return (num > 9) ? num : ('0' + num.toString());
        }

        function getClockTimeForView(dateObj) {
            return makeDoubleCharsNumber(dateObj.getHours()) + ':' + makeDoubleCharsNumber(dateObj.getMinutes());
        }

        function updateClockOnView() {
            $clock.text(getClockTimeForView(clockDateTime));
        }

        function getRemainingSecondsFromClock() {
            return (60 - clockDateTime.getSeconds()) * 1000;
        }

        function updateClockObj(datetime) {
            var prevSeconds = clockDateTime.getSeconds();
            clockDateTime = new Date(datetime);
            if(prevSeconds > clockDateTime.getSeconds()) {
                components.Timezone.updateClock(clockDateTime);
            } else {
                runInterval();
            }
        }

        function runInterval() {
            lastLocalClock = Date.now();
            var remainSecondsFromClock = Math.min(getRemainingSecondsFromClock(), 5000);
            timerId = setTimeout(function () {
                updateClockObj(clockDateTime.valueOf() + (Date.now() - lastLocalClock));
            }, remainSecondsFromClock);
        }

        function scrollPopupContentToCurrentTZ() {
            $currentOption.addClass('selected');

            $scrollableList
                .scrollTop(0) // reset list scroll position
                .scrollTop($currentOption.offset().top - $timezonePopup.offset().top);

            // remove class selected when hovering
            // binded to options and not to list because list includes the scrollbar of itself.
            $options.one('mouseenter.scrollPopup', function() {
                $currentOption.removeClass('selected');
                $options.off('mouseenter.scrollPopup');
            });

        }

        components.Timezone = {
            subscribe: function(fn) {
                if(typeof fn === 'function') callbacks.push(fn);
                return this;
            },
            getTimezone: function() {
                return currentValue;
            },
            getClockEl: function() {
                return $clock;
            },
            setClockAutoUpdate: function(bool) {
                autoUpdateMode = !!bool;
                return this;
            },
            setClock: function(datetimeObj) {
                $clock.text(getClockTimeForView(datetimeObj));
                return this;
            },
            updateClock: function (newDateTimeFormatted) {
                clockDateTime = new Date(newDateTimeFormatted);
                updateClockOnView();
                if (timerId) {
                    clearTimeout(timerId);
                }
                if(autoUpdateMode) {
                    runInterval();
                }
                return this;
            },
            setTimezoneOffset: function(str) {
                $timezoneOffset.html(str);
                return this;
            }
        };

        $("#economicCurrentTime").click(function (eventObject) {
            if (timezonePopup.isShown()) return;
            timezonePopup.show();

            scrollPopupContentToCurrentTZ();

            eventObject.stopPropagation();
        });

        //On selection of a timezone set the hidden select option accordingly.
        //close the selection div. submit the form and reload the events table.
        //update the current time.
        $options
            .click(function () {
                var $this = $(this);
                var optionValue = $this.attr("id").substr(optionsIdsPrefix.length);

                //Select the pressed timeZone
                currentValue = optionValue;
                $currentOption = $this;
                $timezone.val(optionValue);

                //Set the string displaying the time and timezone
                timezonePopup.hide();

                callbacks.forEach(function (fn) {
                    fn(currentValue);
                });
            })
            .each(function () {
                var $this = $(this);
                var optionValue = $this.attr("id").substr(optionsIdsPrefix.length);

                if(currentValue === optionValue) {
                    $currentOption = $this;
                    return false;
                }
            });

        // init clock auto update
        components.Timezone.updateClock(components.Timezone.getClockEl().attr('data-datetime'));

    })(window, jQuery, window.components = window.components || {});
</script>
